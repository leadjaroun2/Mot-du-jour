    </div>

    <script>
        let baseDeDonnees = [];
        fetch('mots.json')
            .then(res => res.json())
            .then(dico => {
                baseDeDonnees = dico;
                tirerUnMot();
            });

        function tirerUnMot() {
            const mot = baseDeDonnees[Math.floor(Math.random() * baseDeDonnees.length)];
            document.getElementById("motFr").innerText = mot.fr;
            document.getElementById("motEn").innerText = mot.en;
            document.getElementById("niveau").innerText = "Niveau: " + mot.niveau;
            document.getElementById("motEn").style.display = "none";
        }

        function reveleTraduction() {
            const el = document.getElementById("motEn");
            if (el.style.display === "none") {
                el.style.display = "block";
            } else {
                tirerUnMot(); // On tire un nouveau mot si on reclique
            }
        }
    </script>
